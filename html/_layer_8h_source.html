299  MatrixXcd A, B;

302 
303  Nx = layerEpsilon.rows(); // # of rows in the epsilon matrix
304  Ny = layerEpsilon.cols(); // # of cols in the epsilon matrix
305 
306  if (fmod(P,2)==0 && fmod(Q,2)==0)
307  {
308  NH = (P+1) * (Q+1);
309  p.resize(P+1);
310  q.resize(Q+1);
311  }
312  else if (fmod(P,2)!=0 && fmod(Q,2)!=0)
313  {
314  NH = P * Q;

316  q.resize(Q);
317  }

332  mu_convmat.resize(NH,NH); // Matrix for convolution
333 
334  // compute indices of spatial harmonics
335  p(0) = -(int)floor(P/2.0);
336  for (int i=1; i<p.size(); i++)
337  p(i) = p(i-1)+1;

358  for (int qrow=0; qrow<Q; qrow++)

361  row = qrow*P + prow;
362  for (int qcol=0; qcol<Q; qcol++)
363  for (int pcol=0; pcol<P; pcol++)
