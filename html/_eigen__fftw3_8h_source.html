141 inline MatrixXcd Eigen_fftw3::fftShift2d(const MatrixXcd &mat)
142 {
143  int m, n, p, q;
144  m = mat.rows();
145  n = mat.cols();
146 
147  // matrix to store fftshift data
148  MatrixXcd mat_fftshift(m,n);
149 
150  // odd # of rows and cols
151  if ((int)fmod(m,2)==1)
152  {
153  p = (int) floor(m/2.0);
154  q = (int) floor(n/2.0);
155  }
156  else // even # of rows and cols
157  {
158  p = (int) ceil(m/2.0);
159  q = (int) ceil(n/2.0);
160  }
161 
162  // vectors to store swap indices
163  RowVectorXi indx(m), indy(n);
164 
165  // compute swap indices
166  if ((int)fmod(m,2)==1) // # of rows odd
167  {
168  for (int i=0; i<m-p-1; i++)
169  indx(i) = (m-p)+i;
170  for (int i=m-p-1; i<m; i++)
171  indx(i) = i-(m-p-1);
172  }
173  else // # of rows even
174  {
175  for (int i=0; i<m-p; i++)
176  indx(i) = p+i;
177  for (int i=m-p; i<m; i++)
178  indx(i) = i-(m-p);
179  }
180 
181  if ((int)fmod(n,2)==1) // # of cols odd
182  {
183  for (int i=0; i<n-q-1; i++)
184  indy(i) = (n-q)+i;
185  for (int i=n-q-1; i<n; i++)
186  indy(i) = i-(n-q-1);
187  }
188  else // # of cols even
189  {
190  for (int i=0; i<n-q; i++)
191  indy(i) = q+i;
192  for (int i=n-q; i<n; i++)
193  indy(i) = i-(n-q);
194  }
195 
196  // rearrange the matrix elements by swapping the elements
197  // according to the indices computed above.
198  for (int i=0; i<m; i++)
199  for (int j=0; j<n; j++)
200  mat_fftshift(i,j) = mat(indx(i),indy(j));
